# cmdk v1.3.0 Commands — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Extend the cmdk palette's `>` scope from pages-only to "do-or-go" by introducing a first-class `CommandItem` registry and shipping 7 stateless global verbs (Upload, Create album, Create shared space, Sign out, Keyboard shortcuts, Toggle theme, Clear palette recents).

**Architecture:** New `command-items.ts` registry parallels `navigation-items.ts`. Two provider registrations (`providers.navigation` + `providers.commands`) share a `filterNavAndCommands` helper, preserving the one-provider-per-section invariant. A new `sections.commands` alongside `sections.navigation` renders via a new `GlobalSearchCommandsSection.svelte`. `kind: 'command'` extends the `ActiveItem` / `activate()` unions with a single-flight guard and no-RECENT-write semantics. The existing `nav:theme` action migrates to `cmd:theme`, and the `category: 'actions'` branch in `activate('nav')` is removed.

**Tech Stack:** Svelte 5 (runes), TypeScript strict, Vitest + @testing-library/svelte (happy-dom), Playwright (E2E), pnpm workspaces, Tailwind CSS 4, `@immich/ui` primitives.

**Source of truth for requirements:** `docs/plans/2026-04-18-cmdk-v1.3-design.md` (committed). Refer back whenever this plan seems under-specified.

**Step 4: Implement the shared match helper**

```ts
// web/src/lib/managers/cmdk-match.ts
const NON_ALNUM = /[^a-z0-9]+/;

/**
 * Shared "almost exact" word-prefix match used by both the navigation and
 * commands providers for top-result promotion under the `'all'` scope.
 *
 * Rule: after case-folding and splitting both sides on non-alphanumerics, at
 * least one query word ≥ minLength chars must be a prefix of some label word.
 */
export function isAlmostExactWordMatch(query: string, label: string, minLength: number): boolean {
  const q = query.trim().toLowerCase();
  if (q.length < minLength) {
    return false;
  }
  const qWords = q.split(NON_ALNUM).filter((w) => w.length >= minLength);
  if (qWords.length === 0) {
    return false;
  }
  const labelWords = label.toLowerCase().split(NON_ALNUM).filter(Boolean);
  for (const qw of qWords) {
    for (const lw of labelWords) {
      if (lw.startsWith(qw)) {
        return true;
      }
    }
  }
  return false;
}
```

**Step 1: Extract the shared filter**

```ts
interface NavAndCommandsResult {
  commands: CommandItem[];
  navigation: NavigationItem[];
}

/**
 * Shared filter path for the nav + commands providers. Applies admin +
 * feature-flag gates to both registries, then returns both filtered arrays
 * for the caller to route into its own section.
 */
private filterNavAndCommands(payload: string, scope: Scope): NavAndCommandsResult {
  if (scope !== 'all' && scope !== 'nav') {
    return { commands: [], navigation: [] };
  }
  if (scope === 'all' && payload === '') {
    return { commands: [], navigation: [] };
  }

  const u = get(user);
  const isAdmin = u?.isAdmin ?? false;
  const flags = featureFlagsManager.valueOrUndefined;

  const eligibleNav: NavigationItem[] = [];
  for (const item of NAVIGATION_ITEMS) {
    if (item.adminOnly && !isAdmin) continue;
    if (item.featureFlag && !flags?.[item.featureFlag]) continue;
    eligibleNav.push(item);
  }

  const eligibleCmd: CommandItem[] = [];
  for (const item of COMMAND_ITEMS) {
    if (item.adminOnly && !isAdmin) continue;
    if (item.featureFlag && !flags?.[item.featureFlag]) continue;
    eligibleCmd.push(item);
  }

  const translate = get(t);

  if (scope === 'nav' && payload === '') {
    const sortByLabel = <T extends { labelKey: string }>(a: T, b: T) =>
      translate(a.labelKey as Translations).localeCompare(translate(b.labelKey as Translations));
    return {
      commands: [...eligibleCmd].sort(sortByLabel),
      navigation: [...eligibleNav].sort(sortByLabel),
    };
  }

  // Non-empty payload. Reuse the existing nav fuzzy search (corpus +
  // computeCommandScore) for nav, and apply the same scoring path to commands.
  // Top-N slice applied by each provider wrapper, not here.
  const navSearch = this.getNavigationSearchStrings();
  const scoredNav: Array<{ item: NavigationItem; score: number }> = [];
  for (const item of eligibleNav) {
    const corpus = navSearch.get(item.id);
    if (!corpus) continue;
    const score = computeCommandScore(corpus, payload);
    if (score <= 0) continue;
    scoredNav.push({ item, score });
  }
  scoredNav.sort((a, b) => b.score - a.score);

  const cmdSearch = this.getCommandSearchStrings();
  const scoredCmd: Array<{ item: CommandItem; score: number }> = [];
  for (const item of eligibleCmd) {
    const corpus = cmdSearch.get(item.id);
    if (!corpus) continue;
    const score = computeCommandScore(corpus, payload);
    if (score <= 0) continue;
    scoredCmd.push({ item, score });
  }
  scoredCmd.sort((a, b) => b.score - a.score);

  return {
    commands: scoredCmd.map((s) => s.item),
    navigation: scoredNav.map((s) => s.item),
  };
}
```

private commandSearchCache = new Map<string, Map<string, string>>();

private getCommandSearchStrings(): Map<string, string> {
  const currentLocale = get(locale);
  const cached = this.commandSearchCache.get(currentLocale);
  if (cached) return cached;

  const translate = get(t);
  const table = new Map<string, string>();
  for (const item of COMMAND_ITEMS) {
    const label = translate(item.labelKey as Translations);
    const description = translate(item.descriptionKey as Translations);
    table.set(item.id, `${label} ${description}`);
  }
  this.commandSearchCache.set(currentLocale, table);
  return table;
}

private runNavigationProvider(payload: string, scope: Scope): ProviderStatus<NavigationItem> {
  const { navigation } = this.filterNavAndCommands(payload, scope);
  if (navigation.length === 0) return { status: 'empty' };
  return { status: 'ok', items: navigation.slice(0, this.providers.navigation.topN), total: navigation.length };
}

private runCommandsProvider(payload: string, scope: Scope): ProviderStatus<CommandItem> {
  const { commands } = this.filterNavAndCommands(payload, scope);
  if (commands.length === 0) return { status: 'empty' };
  return { status: 'ok', items: commands.slice(0, this.providers.commands.topN), total: commands.length };
}
