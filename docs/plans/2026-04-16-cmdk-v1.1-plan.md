# Cmd/Ctrl+K Palette v1.1 Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Extend the v1 cmdk palette (PR #349) with Album + Space sections in the streaming fan-out, and a "× N more" truncation indicator on sections lacking a see-all target.

**Architecture:** Client-side filter over a cached lightweight catalog. New server endpoint `GET /albums/names` avoids the `updateThumbnails()` side-effect. Two abort lifecycles: `closeSignal` (catalog fetch + activation) and per-keystroke `batch.signal` (fan-out). Section truncation is dual-rendered as a visible heading suffix (`ALBUMS (5 of 8)`) + an `aria-hidden` chip row.

**Tech Stack:** NestJS 11 + Kysely (server); SvelteKit + Svelte 5 + Bits UI `Command` (web); `oazapfts` SDK; Vitest + `@testing-library/svelte`; Playwright (E2E).

**Design doc:** `docs/plans/2026-04-16-cmdk-v1.1-design.md` — read for rationale, trade-offs, edge cases. This plan is the "how," not the "why."

```ts
@GenerateSql({ params: [DummyValue.UUID] })
async getOwnedNames(ownerId: string) {
  return this.db
    .selectFrom('album')
    .select([
      'album.id',
      'album.albumName',
      'album.albumThumbnailAssetId',
    ])
    .select((eb) =>
      eb
        .selectFrom('album_asset')
        .select(eb.fn.countAll<number>().as('assetCount'))
        .whereRef('album_asset.albumsId', '=', 'album.id')
        .as('assetCount'),
    )
    .select((eb) =>
      eb
        .selectFrom('album_asset')
        .innerJoin('asset', 'asset.id', 'album_asset.assetsId')
        .select(eb.fn.min('asset.localDateTime').as('startDate'))
        .whereRef('album_asset.albumsId', '=', 'album.id')
        .as('startDate'),
    )
    .select((eb) =>
      eb
        .selectFrom('album_asset')
        .innerJoin('asset', 'asset.id', 'album_asset.assetsId')
        .select(eb.fn.max('asset.localDateTime').as('endDate'))
        .whereRef('album_asset.albumsId', '=', 'album.id')
        .as('endDate'),
    )
    .where('album.ownerId', '=', ownerId)
    .orderBy('album.albumName')
    .execute();
}
```

Note: the repository method intentionally omits `shared` from the SELECT — owned albums are always `shared: false`. The service mapper (Task 3) hardcodes this in the DTO shape. Do not reuse `isActivityEnabled` (which is a completely unrelated field — album activity feed toggle, not sharing state).

```ts
async getNames(auth: AuthDto): Promise<AlbumNameDto[]> {
  const [owned, shared] = await Promise.all([
    this.albumRepository.getOwnedNames(auth.user.id),
    this.albumRepository.getSharedNames(auth.user.id),
  ]);
  return [
    ...owned.map((r) => ({
      ...r,
      shared: false,
      startDate: asDateString(r.startDate),
      endDate: asDateString(r.endDate),
    })),
    ...shared.map((r) => ({

      shared: true,
      startDate: asDateString(r.startDate),
      endDate: asDateString(r.endDate),
    })),
  ];
```

**Step 3: Implement**

```ts
albumsCache: AlbumNameDto[] | undefined = $state(undefined);
spacesCache: SharedSpaceResponseDto[] | undefined = $state(undefined);
private albumsPromise: Promise<void> | undefined;
private spacesPromise: Promise<void> | undefined;

async ensureAlbumsCache(): Promise<void> {
  if (this.albumsCache) return;
  if (!this.albumsPromise) {
    this.albumsPromise = this.fetchAlbumsCatalog();
  }
  return this.albumsPromise;
}

private async fetchAlbumsCatalog(): Promise<void> {
  try {
    const response = await getAlbumNames({ signal: this.closeSignal });
    // Dedupe by id; iterate and keep the shared=true record when a duplicate appears.
    // (Server returns [...owned, ...shared] concatenated; an owned-and-shared-out
    // album appears twice.)
    const byId = new Map<string, AlbumNameDto>();
    for (const record of response) {
      const existing = byId.get(record.id);
      if (!existing || (record.shared && !existing.shared)) {
        byId.set(record.id, record);
      }
    }
    this.albumsCache = [...byId.values()];
  } catch (err) {
    if ((err as Error).name === 'AbortError') return; // palette closed, silent
    this.sections.albums = { status: 'error' };
    throw err;
  }
}

async ensureSpacesCache(): Promise<void> {
  if (this.spacesCache) return;
  if (!this.spacesPromise) {
    this.spacesPromise = this.fetchSpacesCatalog();
  }
  return this.spacesPromise;
}

private async fetchSpacesCatalog(): Promise<void> {
  try {
    this.spacesCache = await getAllSpaces({ signal: this.closeSignal });
  } catch (err) {
    if ((err as Error).name === 'AbortError') return;
    this.sections.spaces = { status: 'error' };
    throw err;
  }
}
```

Note: `open()` must also clear `albumsPromise` / `spacesPromise` on the abort-then-reopen path so a stale rejected promise isn't returned by `ensureAlbumsCache` in the new session. Extend `open()` accordingly.

```ts
async runAlbums(rawQuery: string) {
  const query = rawQuery.trim().toLowerCase();
  if (query.length < 2) {
    this.sections.albums = { status: 'idle' };
    return;
  }
  await this.ensureAlbumsCache();
  if (!this.albumsCache) return;

  const matches = this.albumsCache
    .map((a) => {
      const name = a.albumName.toLowerCase();
      if (!name.includes(query)) return null;
      const score = name.startsWith(query) ? 2 : 1;
      return { album: a, score };
    })
    .filter((m): m is { album: AlbumNameDto; score: number } => m !== null)
    .sort((a, b) => b.score - a.score || a.album.albumName.localeCompare(b.album.albumName));

  this.sections.albums = {
    status: 'ok',
    items: matches.slice(0, TOP_N).map((m) => m.album),
    total: matches.length,
  };
}
```

```ts
// Create user + seed N albums with given names.
export async function seedAlbums(
  api: ReturnType<typeof createApi>,
  names: string[],
): Promise<AlbumResponseDto[]> {
  return Promise.all(names.map((name) => api.albums.createAlbum({ createAlbumDto: { albumName: name } })));
}

// Create user + seed spaces with given names + owner + optional members.
export async function seedSpaces(
  api: ReturnType<typeof createApi>,
  names: string[],
): Promise<SharedSpaceResponseDto[]> {
  return Promise.all(
    names.map((name) => api.sharedSpaces.createSpace({ createSharedSpaceDto: { name, color: 'Primary' } })),
  );
}

// Compound setup: one user, 2 albums named "Hawaii Beach" / "Hawaii Mountains", 2 spaces named similarly.
export async function setupUserWithAlbumsAndSpaces(ctx: BrowserContext): Promise<{ user: UserDto; api: ReturnType<typeof createApi> }> {
  const { user, api } = await createUserAndLogin(ctx);
  await seedAlbums(api, ['Hawaii Beach', 'Hawaii Mountains']);
  await seedSpaces(api, ['Hawaii Family', 'Hawaii Friends']);
  return { user, api };
}

// Seed the RECENT store with a space id that doesn't resolve.
export async function seedRecentWithNonexistentSpace(page: Page): Promise<void> {
  await page.evaluate(() => {
    const userId = /* read from session store */;
    const entry = {
      kind: 'space',
      id: 'space:nonexistent',
      spaceId: 'nonexistent',
      label: 'Ghost Space',
      colorHex: null,
      lastUsed: Date.now(),
    };
    localStorage.setItem(`cmdk.recent:${userId}`, JSON.stringify([entry]));
  });
}

// Seed one match in every entity section type (photos, albums, spaces, people, places, tags).
export async function seedAllSectionTypes(
  ctx: BrowserContext,
  query: string = 'testmatch',
): Promise<{ user: UserDto; api: ReturnType<typeof createApi> }> {
  const { user, api } = await createUserAndLogin(ctx);
  await api.photos.upload({ filename: `${query}-photo.jpg` });
  await seedAlbums(api, [`${query} Album`]);
  await seedSpaces(api, [`${query} Space`]);
  await api.people.create({ name: `${query} Person` });
  // places: derived from photo GPS metadata — seed via photo EXIF
  // tags: create a tag matching the query
  await api.tags.create({ name: `${query}-tag` });
  return { user, api };
}

// Create album owned by user A + share to user B.
export async function createAndShareAlbum(
  owner: { api: ReturnType<typeof createApi> },
  buddy: { user: UserDto },
  albumName: string,
): Promise<AlbumResponseDto> {
  const album = await owner.api.albums.createAlbum({ createAlbumDto: { albumName } });
  await owner.api.albums.addUsers({
    id: album.id,
    addUsersDto: { albumUsers: [{ userId: buddy.user.id, role: 'viewer' }] },
  });
  return album;
}
```
